expression  -> literal
            |  unary
            |  binary
            |  grouping ;

literal     -> NUMBER | STRING | "true" | "false" | "void" ;

grouping    -> "(" expression ")" ;

unary       -> ( "+" | "-" | "!" ) expression ;

binary      -> expression operator expression ;

operator    -> "==" | "!=" | "<" | "<=" | ">" | ">="
            |  "+"  | "-"  | "*" | "/" ;
